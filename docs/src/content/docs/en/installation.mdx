import { Tabs, TabItem, Steps, LinkButton, Aside } from '@astrojs/starlight/components';

<Aside>
  Edge, Brave, Opera, Opera GX, and Vivaldi are all built on the same base as Chrome, Chromium. Therefore, the installation process is the same as Chrome with these browsers.

  Librewolf, Waterfox, and other Firefox-based browsers *should* also work with the Firefox steps, however, they are not officially supported.
</Aside>

<Tabs>
  <TabItem label="Chrome / Chromium">
    <Steps>
      <ol>
        <li>
          Navigate to the extension on the Chrome Web Store.  
          <LinkButton icon="external" href="https://chromewebstore.google.com/detail/cruggs-anisearch-utils/eanhcibbgmgechngeeihlkijakallpeb" target="_blank">Chrome Web Store</LinkButton>
        </li>
        <li>
          Click the "Add to Chrome" button.
        </li>
        <li>
          Congratulations! You have successfully installed the extension. You can now continue to the [Getting Started](/en/getting-started) guide.
        </li>
      </ol>
    </Steps>
  </TabItem>

  <TabItem label="Firefox">
    <Steps>
      <ol>
        <li>
          Navigate to the extension on the Firefox Add-ons website.  
          <LinkButton icon="external" href="https://addons.mozilla.org/firefox/addon/cruggs-asu/" target="_blank">Firefox Add-ons</LinkButton>
        </li>
        <li>
          Click the "Add to Firefox" button.
        </li>
        <li>
          Congratulations! You have successfully installed the extension. You can now continue to the [Getting Started](/en/getting-started) guide.
        </li>
      </ol>
    </Steps>
  </TabItem>
</Tabs>
