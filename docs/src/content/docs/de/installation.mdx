import { Tabs, TabItem, Steps, LinkButton, Aside } from '@astrojs/starlight/components';

<Aside>
  Edge, Brave, Opera, Opera GX und Vivaldi basieren alle auf der gleichen basis wie Chrome. Der Installationsprozess sollte daher dort der gleiche sein.

  Librewolf, Waterfox und andere Firefox-basierte Browser *sollten* auch funktionieren, jedoch werden sie offiziell nicht unterstützt.
</Aside>

<Tabs>
  <TabItem label="Chrome / Chromium">
    <Steps>
      <ol>
        <li>
          Öffne die Erweiterung im Chrome Web Store.
          <LinkButton icon="external" href="https://chromewebstore.google.com/detail/cruggs-anisearch-utils/eanhcibbgmgechngeeihlkijakallpeb" target="_blank">Chrome Web Store</LinkButton>
        </li>
        <li>
          Klicke auf den "Zu Chrome hinzufügen" Button.
        </li>
        <li>
          Glückwunsch, du hast die Erweiterung erfolgreich installiert. Du kannst nun mit der [Einführung](/de/getting-started) fortfahren.
        </li>
      </ol>
    </Steps>
  </TabItem>

  <TabItem label="Firefox">
    <Steps>
      <ol>
        <li>
          Öffne die Erweiterung auf der Firefox Add-ons Webseite.
          <LinkButton icon="external" href="https://addons.mozilla.org/firefox/addon/cruggs-asu/" target="_blank">Firefox Add-ons</LinkButton>
        </li>
        <li>
          Klicke auf den "Zu Firefox hinzufügen" Button.
        </li>
        <li>
          Glückwunsch! Du hast die Erweiterung erfolgreich installiert. Du kannst nun mit der [Einführung](/de/getting-started) fortfahren.
        </li>
      </ol>
    </Steps>
  </TabItem>
</Tabs>
